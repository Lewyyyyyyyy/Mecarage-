% —— Macros diagrammes de séquence (TikZ pur) ——————————————
% \sdpart{x}{nodevar}{Label} — participant + lifeline (hauteur fixe 9cm)
\newcommand{\sdpart}[3]{%
  \node[sdpart] (#2) at (#1, 0) {\strut #3};
  \draw[draw=diagborder,dashed,thin] (#1,-0.3) -- (#1,-9.5);}
% \sdmsg{from-x}{to-x}{y}{label} — message synchrone
\newcommand{\sdmsg}[4]{%
  \draw[sdmsgstyle] (#1,#3) -- (#2,#3)
    node[midway,above,font=\tiny,fill=white,inner sep=1pt,align=center]{#4};}
% \sdrtn{from-x}{to-x}{y}{label} — retour
\newcommand{\sdrtn}[4]{%
  \draw[sdrtnstyle] (#1,#3) -- (#2,#3)
    node[midway,above,font=\tiny,fill=white,inner sep=1pt,align=center]{#4};}
% \sdself{x}{y}{label} — message auto
\newcommand{\sdself}[3]{%
  \draw[sdmsgstyle] (#1,#2) -- ++(0.6,0) -- ++(0,-0.3) -- ++(-0.6,0)
    node[midway,right=0.6cm,font=\tiny,align=left,fill=white,inner sep=1pt]{#3};}
% \sdbox{x}{ytop}{ybot} — boîte d'activation
\newcommand{\sdbox}[3]{%
  \filldraw[fill=gray!15,draw=diagborder,thin] (#1-0.08,#2) rectangle (#1+0.08,#3);}
